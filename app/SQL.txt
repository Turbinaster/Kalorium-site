WITH p AS (
    SELECT
        TIMESTAMPTZ '2026-03-02 00:00:00+03' AS dt_from,
        now() AS dt_to
),

-- Когорта "открыли приложение в период" (по first_seen) среди тех, у кого есть trial
trial_population AS (
    SELECT d.id AS device_ref_id
    FROM devices d
    JOIN device_trials dt
      ON dt.device_id = d.id
    CROSS JOIN p
    WHERE d.first_seen_at >= p.dt_from
      AND d.first_seen_at <  p.dt_to
),

-- События usage за период только по этой когорте
events_in_period AS (
    SELECT
        ue.device_ref_id,
        ue.source,
        ue.result,
        COALESCE(ue.units, 1) AS units
    FROM usage_events ue
    JOIN trial_population tp
      ON tp.device_ref_id = ue.device_ref_id
    CROSS JOIN p
    WHERE ue.occurred_at >= p.dt_from
      AND ue.occurred_at <  p.dt_to
),

-- Метрики на устройство из когорты
per_device AS (
    SELECT
        tp.device_ref_id,
        COALESCE(SUM(
            CASE
                WHEN e.source IN ('subscription', 'trial') AND e.result = 'ok'
                THEN e.units ELSE 0
            END
        ), 0) AS paid_ok_requests,
        COALESCE(SUM(
            CASE
                WHEN e.result = 'ok'
                THEN e.units ELSE 0
            END
        ), 0) AS any_ok_requests
    FROM trial_population tp
    LEFT JOIN events_in_period e
      ON e.device_ref_id = tp.device_ref_id
    GROUP BY tp.device_ref_id
),

active AS (
    SELECT device_ref_id, paid_ok_requests
    FROM per_device
    WHERE paid_ok_requests >= 1
),
active_total AS (
    SELECT COUNT(*)::numeric AS cnt FROM active
),
buckets AS (
    SELECT generate_series(1, 10) AS bucket_n
),

distribution_1_10 AS (
    SELECT
        'paid_ok_requests_distribution_1_10'::text AS section,
        b.bucket_n::text AS bucket,
        COUNT(a.device_ref_id)::bigint AS users_count,
        ROUND(
            100.0 * COUNT(a.device_ref_id)::numeric / NULLIF(t.cnt, 0),
            2
        ) AS users_percent
    FROM buckets b
    CROSS JOIN active_total t
    LEFT JOIN active a
      ON a.paid_ok_requests = b.bucket_n
    GROUP BY b.bucket_n, t.cnt
),

summary AS (
    SELECT
        'summary'::text AS section,
        'trial_users_opened_in_period'::text AS bucket,
        COUNT(*)::bigint AS users_count,
        NULL::numeric AS users_percent
    FROM per_device

    UNION ALL

    SELECT
        'summary'::text AS section,
        'opened_but_no_ok_requests'::text AS bucket,
        COUNT(*)::bigint AS users_count,
        ROUND(
            100.0 * COUNT(*)::numeric / NULLIF((SELECT COUNT(*) FROM per_device), 0),
            2
        ) AS users_percent
    FROM per_device
    WHERE any_ok_requests = 0
),

-- Устройства из "opened_but_no_ok_requests"
opened_but_no_ok AS (
    SELECT device_ref_id
    FROM per_device
    WHERE any_ok_requests = 0
),

-- Признак "есть хотя бы один привязанный user с заполненным профилем"
profile_by_device AS (
    SELECT
        o.device_ref_id,
        COALESCE(
            BOOL_OR(
                u.profile_sex IS NOT NULL
                AND u.profile_age IS NOT NULL
                AND u.profile_height_cm IS NOT NULL
                AND u.profile_weight_kg IS NOT NULL
                AND u.profile_activity IS NOT NULL
                AND u.profile_goal IS NOT NULL
            ),
            FALSE
        ) AS has_full_profile
    FROM opened_but_no_ok o
    LEFT JOIN user_devices ud
      ON ud.device_id = o.device_ref_id
    LEFT JOIN users u
      ON u.id = ud.user_id
    GROUP BY o.device_ref_id
),

opened_no_ok_totals AS (
    SELECT
        COUNT(*)::numeric AS opened_no_ok_cnt
    FROM opened_but_no_ok
),

additional_profile_section AS (
    SELECT
        'opened_but_no_ok_profile'::text AS section,
        'opened_no_ok_without_full_profile'::text AS bucket,
        COUNT(*)::bigint AS users_count,
        ROUND(
            100.0 * COUNT(*)::numeric / NULLIF((SELECT opened_no_ok_cnt FROM opened_no_ok_totals), 0),
            2
        ) AS users_percent
    FROM profile_by_device
    WHERE has_full_profile = FALSE

    UNION ALL

    SELECT
        'opened_but_no_ok_profile'::text AS section,
        'opened_no_ok_without_full_profile_from_all_opened'::text AS bucket,
        COUNT(*)::bigint AS users_count,
        ROUND(
            100.0 * COUNT(*)::numeric / NULLIF((SELECT COUNT(*) FROM per_device), 0),
            2
        ) AS users_percent
    FROM profile_by_device
    WHERE has_full_profile = FALSE
)

SELECT section, bucket, users_count, users_percent
FROM (
    SELECT
        section, bucket, users_count, users_percent,
        1 AS section_order,
        bucket::int AS bucket_order
    FROM distribution_1_10

    UNION ALL

    SELECT
        section, bucket, users_count, users_percent,
        2 AS section_order,
        CASE
            WHEN bucket = 'trial_users_opened_in_period' THEN 1
            WHEN bucket = 'opened_but_no_ok_requests' THEN 2
            ELSE 999
        END AS bucket_order
    FROM summary

    UNION ALL

    SELECT
        section, bucket, users_count, users_percent,
        3 AS section_order,
        CASE
            WHEN bucket = 'opened_no_ok_without_full_profile' THEN 1
            WHEN bucket = 'opened_no_ok_without_full_profile_from_all_opened' THEN 2
            ELSE 999
        END AS bucket_order
    FROM additional_profile_section
) x
ORDER BY section_order, bucket_order;
